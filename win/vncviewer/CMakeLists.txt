	${CMAKE_SOURCE_DIR}/common/d3des ${CMAKE_CURRENT_BINARY_DIR}/res
	${CMAKE_SOURCE_DIR}/win/util)

	set(APPNAME "TurboVNC Viewer 64-bit v${VERSION} (${BUILD})")

	install(TARGETS cvncviewer DESTINATION ${CMAKE_INSTALL_BINDIR})
